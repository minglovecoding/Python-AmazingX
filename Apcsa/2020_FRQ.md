public class Hailstone
{
    public static int hailstoneLength(int n)
    {
        int count = 1;

        while (n != 1)
        {
            if (n % 2 == 0)
            {
                n = n / 2;
            }
            else
            {
                n = 3 * n + 1;
            }

            count++;
        }

        return count;
    }

    public static boolean isLongSeq(int n)
    {
        return hailstoneLength(n) > n;
    }

    public static double propLong(int n)
    {
        int count = 0;

        for (int i = 1; i <= n; i++)
        {
            if (isLongSeq(i))
            {
                count++;
            }
        }

        return (double) count / n;
    }
}

public class GameSpinner
{
    private int sectors;
    private int current;
    private int run;

    public GameSpinner(int numSectors)
    {
        sectors = numSectors;
        current = 0;
        run = 0;
    }

    public int spin()
    {
        int result = (int)(Math.random() * sectors) + 1;

        if (result == current)
        {
            run++;
        }
        else
        {
            current = result;
            run = 1;
        }

        return result;
    }

    public int currentRun()
    {
        return run;
    }
}

{
